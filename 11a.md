## start mysql
```bash
docker run --name mysql-test -e MYSQL_ROOT_PASSWORD=root -d -p 3306:3306 mysql
```
---
## enter mysql
```bash
docker exec -it mysql-test mysql -uroot -proot
```
---

```sql
CREATE DATABASE testdb;
USE testdb;

CREATE TABLE users (
  id VARCHAR(10) PRIMARY KEY,
  name VARCHAR(50),
  age INT
```
